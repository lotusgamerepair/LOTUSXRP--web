    <h1>Lotus Flower</h1>

    <input id="usage" type="number" placeholder="Enter gallons used today" min="0">

    <button onclick="updateLotus()">Submit</button>

    <div id="lotus" style="height: 20px; background-color: grey;"></div>

    <p id="message">Track your water usage!</p>

    <script>

        function updateLotus() {

            let gallons = parseInt(document.getElementById("usage").value);

            let lotus = document.getElementById("lotus");

            let msg = document.getElementById("message");

            if (isNaN(gallons) || gallons < 0) {

                lotus.style.height = "20px";

                lotus.style.backgroundColor = "grey";

                msg.textContent = "Please enter a valid number!";

                return;

            }

            if (gallons < 30) {

                lotus.style.height = "100px";

                lotus.style.backgroundColor = "green";

                msg.textContent = `You used ${gallons} gallons—great job!`;

            } else if (gallons <= 60) {

                lotus.style.height = "50px";

                lotus.style.backgroundColor = "yellow";

                msg.textContent = `You used ${gallons} gallons—aim lower tomorrow!`;

            } else {

                lotus.style.height = "20px";

                lotus.style.backgroundColor = "red";

                msg.textContent = `You used ${gallons} gallons—time to save more!`;

            }

        }

    </script>
